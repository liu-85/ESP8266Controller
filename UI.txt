    <script>
        // 主题切换
        const themes = ['theme-1', 'theme-2', 'theme-3', 'theme-4'];
        document.querySelectorAll('.theme-btn').forEach(btn => {
            btn.addEventListener('click', () => {
                document.querySelectorAll('.theme-btn').forEach(b => b.classList.remove('active'));
                btn.classList.add('active');
                document.body.className = btn.dataset.theme;
            });
        });

        // 主图标元素
        const mainWifiIcon = document.getElementById('mainWifiIcon');
        const mainBtIcon = document.getElementById('mainBtIcon');

        // 连接方式切换
        const btModeBtn = document.getElementById('btModeBtn');
        const wifiModeBtn = document.getElementById('wifiModeBtn');
        const wifiSettings = document.getElementById('wifiSettings');
        const connectionStatus = document.getElementById('connectionStatus');

        // 切换WIFI
        wifiModeBtn.addEventListener('click', () => {
            wifiModeBtn.classList.add('active');
            btModeBtn.classList.remove('active');
            wifiSettings.style.display = 'block';
            connectionStatus.textContent = 'WIFI未连接';
            
            // 图标显示WIFI 隐藏蓝牙
            mainWifiIcon.classList.add('show');
            mainBtIcon.classList.remove('show');
            mainWifiIcon.classList.remove('connected');
        });

        // 切换蓝牙
        btModeBtn.addEventListener('click', () => {
            btModeBtn.classList.add('active');
            wifiModeBtn.classList.remove('active');
            wifiSettings.style.display = 'none';
            connectionStatus.textContent = '蓝牙未连接';
            
            // 图标显示蓝牙 隐藏WIFI
            mainBtIcon.classList.add('show');
            mainWifiIcon.classList.remove('show');
            mainBtIcon.classList.remove('connected');
        });

        // 保存连接
        document.getElementById('saveConnectionBtn').addEventListener('click', () => {
            const ip = document.getElementById('ipInput').value;
            const port = document.getElementById('portInput').value;
            if (wifiModeBtn.classList.contains('active')) {
                connectionStatus.textContent = `已保存 WIFI: ${ip}:${port}`;
                mainWifiIcon.classList.add('connected');
                mainBtIcon.classList.remove('connected');
            } else {
                connectionStatus.textContent = '蓝牙已连接（模拟）';
                mainBtIcon.classList.add('connected');
                mainWifiIcon.classList.remove('connected');
            }
        });

        // 打开/关闭设置面板
        document.getElementById('openSettings').addEventListener('click', () => {
            document.getElementById('settingsPanel').style.display = 'flex';
        });
        document.getElementById('closeSettings').addEventListener('click', () => {
            document.getElementById('settingsPanel').style.display = 'none';
        });

        // 按钮/开关切换状态
        document.querySelectorAll('.btn').forEach(btn => {
            if (btn.id !== 'openSettings') {
                btn.addEventListener('click', () => {
                    btn.classList.toggle('active');
                });
            }
        });
        document.querySelectorAll('.switch').forEach(sw => {
            sw.addEventListener('click', () => {
                sw.classList.toggle('active');
            });
        });

        // 状态变量
        let gyroEnabled = false;
        let timerEnabled = false;
        let gyroX = 0, gyroY = 0;

        // 陀螺仪开关
        const gyroToggle = document.getElementById('gyroToggle');
        gyroToggle.addEventListener('click', () => {
            gyroEnabled = !gyroEnabled;
            gyroToggle.textContent = `陀螺仪: ${gyroEnabled ? 'ON' : 'OFF'}`;
            gyroToggle.classList.toggle('active', gyroEnabled);

            if (gyroEnabled) {
                if (typeof DeviceOrientationEvent !== 'undefined' && 
                    typeof DeviceOrientationEvent.requestPermission === 'function') {
                    DeviceOrientationEvent.requestPermission()
                        .then(response => {
                            if (response === 'granted') {
                                window.addEventListener('deviceorientation', handleOrientation);
                            }
                        })
                        .catch(console.error);
                } else {
                    window.addEventListener('deviceorientation', handleOrientation);
                }
            } else {
                window.removeEventListener('deviceorientation', handleOrientation);
                resetJoystick('leftKnob');
            }
        });

        // 定时发送开关
        const timerToggle = document.getElementById('timerToggle');
        timerToggle.addEventListener('click', () => {
            timerEnabled = !timerEnabled;
            timerToggle.textContent = `定时发送: ${timerEnabled ? 'ON' : 'OFF'}`;
            timerToggle.classList.toggle('active', timerEnabled);
        });

        // 处理陀螺仪数据
        function handleOrientation(event) {
            const gamma = event.gamma || 0;
            const beta = event.beta || 0;
            gyroX = Math.max(-1, Math.min(1, gamma / 45));
            gyroY = Math.max(-1, Math.min(1, (beta - 45) / 45));
        }

        // 摇杆控制逻辑
        function setupJoystick(containerId, knobId, useGyro = false) {
            const container = document.getElementById(containerId);
            const knob = document.getElementById(knobId);
            const radius = container.offsetWidth / 2;
            const knobRadius = knob.offsetWidth / 2;
            const maxDist = radius - knobRadius;

            let isDragging = false;
            let centerX = radius;
            let centerY = radius;

            function moveKnob(x, y) {
                let dx = x - centerX;
                let dy = y - centerY;
                const dist = Math.sqrt(dx*dx + dy*dy);
                if (dist > maxDist) {
                    const ratio = maxDist / dist;
                    dx *= ratio;
                    dy *= ratio;
                }
                knob.style.left = `${centerX + dx}px`;
                knob.style.top = `${centerY + dy}px`;

                const throttle = Math.round((-dy / maxDist) * 100);
                const steering = Math.round((dx / maxDist) * 100);
                updateStatus(`${knobId === 'leftKnob' ? '左' : '右'}摇杆: T=${throttle}, S=${steering}`);
            }

            function reset() {
                knob.style.left = `${centerX}px`;
                knob.style.top = `${centerY}px`;
            }

            container.addEventListener('touchstart', e => {
                if (useGyro && gyroEnabled) return;
                isDragging = true;
            });
            container.addEventListener('touchmove', e => {
                if (!isDragging || (useGyro && gyroEnabled)) return;
                const touch = e.touches[0];
                const rect = container.getBoundingClientRect();
                moveKnob(touch.clientX - rect.left, touch.clientY - rect.top);
            });
            container.addEventListener('touchend', reset);

            if (useGyro) {
                function gyroLoop() {
                    if (gyroEnabled) {
                        moveKnob(centerX + gyroX * maxDist, centerY + gyroY * maxDist);
                    }
                    requestAnimationFrame(gyroLoop);
                }
                gyroLoop();
            }
        }

        function resetJoystick(knobId) {
            const knob = document.getElementById(knobId);
            const container = knob.parentElement;
            const centerX = container.offsetWidth / 2;
            const centerY = container.offsetHeight / 2;
            knob.style.left = `${centerX}px`;
            knob.style.top = `${centerY}px`;
        }

        function updateStatus(text) {
            document.getElementById('status').textContent = text;
        }

        setupJoystick('leftJoystick', 'leftKnob', true);
        setupJoystick('rightJoystick', 'rightKnob', false);

        window.addEventListener('resize', () => {
            resetJoystick('leftKnob');
            resetJoystick('rightKnob');
        });
    </script>
